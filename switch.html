Switch Numbers

let day=1;
switch(day){
 case 1: document.write("Monday"); break;
 case 2: document.write("Tuesday"); break;
 default: document.write("Other");
}

Switch Strings

let role="admin";
switch(role){
 case "admin": document.write("Full Access"); break;
 case "user": document.write("Limited Access"); break;
 default: document.write("Guest");
}
